# 经典排序算法
## 归并排序
写法一：
```
func merge(a, b []int) []int {
	var i, j int
	ret := []int{}
	for i < len(a) && j < len(b) {
		if a[i] <= b[j] {
			ret = append(ret, a[i])
			i++
		} else {
			ret = append(ret, b[j])
			j++
		}
	}
	ret = append(ret, a[i:]...)
	ret = append(ret, b[j:]...)
	return ret
}

func mergeSort(nums []int) []int {
	length := len(nums)
	if length < 2 {
		return nums
	}
	left := mergeSort(nums[:length/2])
	right := mergeSort(nums[length/2:])
	return merge(left, right)
}
```
